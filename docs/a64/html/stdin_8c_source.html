34static int stdin_get_char_nonblocking(void)
35{
36 fd_set rfds;
37 struct timeval tv = {0, 0}; // zero timeout → non-blocking
38
39 FD_ZERO(&rfds);
40 FD_SET(STDIN_FILENO, &rfds);
41
42 int retval = select(STDIN_FILENO + 1, &rfds, NULL, NULL, &tv);
43
44 if (retval <= 0)
45 return -1; // no input
46
47 unsigned char ch;
48 ssize_t n = read(STDIN_FILENO, &ch, 1);
49
50 if (n <= 0)
51 return -1;
52
53 return ch;
54}
