26void input_mcal_init(void)
27{
28 /* External interrupt 0 (PD2) for push-buttons */
29 EICRA &= (uint8_t)~((1 << ISC00) | (1 << ISC01)); // Low-level trigger
30 EIFR |= (uint8_t)(1 << INTF0); // Clear pending flag

37}

64ISR(INT0_vect)
65{
66 /* Snapshot of PORTA */
67 uint8_t pa = PINA;
68
69 /* Disable INT0 */
70 EIMSK &= (uint8_t)~(1 << INT0);
71 EIFR |= (uint8_t)(1 << INTF0);
72
73 /* Decode PA7:5 = 0b1xx */
74 uint8_t code = (pa >> 5) & 0x07;
75 switch (code)
76 {
77 case 0x04:
78 int0_key = HW_KEY_CANCEL; // Lower push-button
79 break;
80 case 0x05:
81 int0_key = HW_KEY_BACK; // Middle push-button
82 break;
83 case 0x06:
84 int0_key = HW_KEY_SELECT; // Upper push-button
85 break;
86 case 0x07:
87 int0_key = HW_KEY_UNUSED; // Unused slot
88 break;
89 default:
90 int0_key = HW_KEY_INVALID; // 0b0xx -> PA7=0 -> switches electrically disabled
91 break;
92 }
93
94 /* Start timer 0 */
95 start_timer0_if_idle();
96}

98ISR(PCINT3_vect)
99{
100 static uint8_t last_state = 0; // 2-bit: ((A << 1) | B)
101
102 uint8_t pd = PIND;
103 uint8_t rota = (pd >> PD5_ROTA) & 0x01;
104 uint8_t rotb = (pd >> PD4_ROTB) & 0x01;
105
106 uint8_t new_state = (uint8_t)((rota << 1) | rotb);
107
108 /* Rotation direction determined when lines A and B return to low */
109 if (new_state == 0 && last_state != 0)
110 {
111 if (last_state == 2) // 0b10 -> 0b00 => A leads
112 {
113 isr_key = HW_KEY_ROTA;
114 }
115 else if (last_state == 1) // 0b01 -> 0b00 => B leads
116 {
117 isr_key = HW_KEY_ROTB;
118 }
119 else
120 {
121 isr_key = HW_KEY_INVALID;
122 }
123 }
124
125 last_state = new_state;
126}
