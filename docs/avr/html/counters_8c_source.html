20void counters_read(uint32_t count[2])
21{
22 // Error conditions
23 if (DDRC != 0x00) error_fatal(); // Port C should be input (default)
24 if ((PINA & (1 << PA7)) != 0) error_fatal(); // PA7 should be low (default)
25 if ((PINB & (1 << PB3_RCLK)) != 0) error_fatal(); // RCLK should be low (default)
26 if ((PINB & (1 << PB0_nEXEC)) == 0) error_fatal(); // EXEC should be low (!EXEC high) (default)
27
28 uint8_t counter_num;
29 uint8_t mask;
30 uint8_t value;
31
32 // Toggle RCLK
33 mask = (1 << PB3_RCLK);
34 value = (1 << PB3_RCLK); // RCLK high;
35 gpio_write_masked(mask, value, GPIO_PORTB);
36 _delay_us(1); // spec. 10 ns
37 value = (0 << PB3_RCLK); // RCLK low;
38 gpio_write_masked(mask, value, GPIO_PORTB);
39
40 // Read counters 3,2,1,0 into count[0] and 7,6,5,4 into count[1]
41 for (int j = 0; j < 2; ++j)
42 {
43 count[j] = 0;
44 for (int i = 3; i >= 0; --i)
45 {
46 // Select and enable counter
47 counter_num = i + (j * 4);
48 counter_num &= 0x07; // Enforce range 0-7 (0b000 to 0b111)
49 mask = (1 << PA0) | (1 << PA1) | (1 << PA2) | (1 << PA7);
50 value = (counter_num << PA0) | (1 << PA7); // Counter number in PA0-PA2, PA7 enabled
51 gpio_write_masked(mask, value, GPIO_PORTA);
52
53 // Toggle EXEC to output latch value
54 mask = (1 << PB0_nEXEC);
55 value = (0 << PB0_nEXEC); // EXEC high (!EXEC low);
56 gpio_write_masked(mask, value, GPIO_PORTB);
57
58 // Insert 8-bit counter value into 32-bit count value
59 count[j] = (count[j] << 8); // Shift count 8 bits left
60 count[j] += PINC; // Add 8-bit counter value
61
62 // Reset EXEC
63 value = (1 << PB0_nEXEC); // EXEC low (!EXEC high);
64 gpio_write_masked(mask, value, GPIO_PORTB);
65 }
66 }
67
68 // Reset defaults
69 mask = (1 << PA7_ENABLE);
70 value = (0 << PA7_ENABLE);
71 gpio_write_masked(mask, value, GPIO_PORTA);
72}
