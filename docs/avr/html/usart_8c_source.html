30void serial_transmit(uint8_t data)
31{
32 // Wait for empty transmit buffer
33 while (!(UCSR0A & (1 << UDRE0))); // Buffer can be written when UDREn flag in UCSRnA register is set
34

39uint8_t serial_receive(void)
40{
41 // USART receiver error flags
42 if (UCSR0A & (1 << FE0)) error_fatal(); // Frame error
43 if (UCSR0A & (1 << DOR0)) error_fatal(); // Data over-run
44 if (UCSR0A & (1 << UPE0)) error_fatal(); // Parity error
45
46 // Wait for data to be received
47 while (!(UCSR0A & (1 << RXC0)));
48
